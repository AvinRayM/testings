Quick Start Guide

1
Open your terminal in the backend/ folder and run: 
npm init -y
npm install express
node server.js

Then open http://localhost:3000 in your browser. That's this page!

2
In Roblox Studio, click on ServerScriptService in the Explorer, then in the Properties panel, check LoadStringEnabled = true.
3
In Roblox Studio, go to Game Settings > Security and enable Allow HTTP Requests.
4
Create a Script inside ServerScriptService and paste the Server Script below.
5
Create a LocalScript inside StarterPlayerScripts (or a GUI) and paste the LocalScript below.
6
Play your game. The server will poll this web panel every 5 seconds. Push a module from the Execute tab and watch it run!
Important: Replace YOUR_URL_HERE in the Roblox scripts with your actual hosted backend URL (e.g., https://your-app.glitch.me). If testing locally, Roblox Studio cannot reach localhost — you must use a tunnel like ngrok or host it online.

ServerScript → ServerScriptService

-- ============================================
-- SERVER SCRIPT (ServerScriptService)
-- Polls the web panel + listens to RemoteEvent
-- ============================================

local HttpService = game:GetService("HttpService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

-- CHANGE THIS to your hosted backend URL
local API_URL = "YOUR_URL_HERE"

-- Create RemoteEvent for LocalScript communication
local execRemote = Instance.new("RemoteEvent")
execRemote.Name = "ExecuteRemote"
execRemote.Parent = ReplicatedStorage

-- Admin whitelist (add your UserId here)
local ADMINS = {
    [123456789] = true,  -- Replace with your actual UserId
}

local function executeCode(codeStr)
    local func, compileErr = loadstring(codeStr)
    if not func then
        return false, "Compile Error: " .. tostring(compileErr)
    end
    local success, runErr = pcall(func)
    if not success then
        return false, "Runtime Error: " .. tostring(runErr)
    end
    return true, "Success"
end

-- Listen for in-game RemoteEvent from LocalScript
execRemote.OnServerEvent:Connect(function(player, codeStr)
    if not ADMINS[player.UserId] then
        warn("Blocked execution from non-admin: " .. player.Name)
        return
    end
    print("[Executor] Running code from " .. player.Name)
    local ok, result = executeCode(codeStr)
    if not ok then warn("[Executor] " .. result) end
end)

-- Poll the web backend every 5 seconds
print("[Web Executor] Server started, polling " .. API_URL)
task.spawn(function()
    while task.wait(5) do
        local ok, res = pcall(function()
            return HttpService:GetAsync(API_URL .. "/api/poll")
        end)

        if ok and res then
            local data = HttpService:JSONDecode(res)

            if data and data.jobId and data.script then
                print("[Web Executor] Got job #" .. data.jobId .. ": " .. data.script)

                local execOk, execMsg = executeCode(data.script)

                pcall(function()
                    HttpService:PostAsync(
                        API_URL .. "/api/result",
                        HttpService:JSONEncode({
                            jobId = data.jobId,
                            status = execOk and "Completed" or "Error",
                            error = execMsg
                        }),
                        Enum.HttpContentType.ApplicationJson,
                        false
                    )
                end)
            end
        end
    end
end)

LocalScript → StarterPlayerScripts or inside a GUI

-- ============================================
-- LOCAL SCRIPT (StarterPlayerScripts or GUI)
-- Fires the RemoteEvent so the Server runs it
-- ============================================

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local execRemote = ReplicatedStorage:WaitForChild("ExecuteRemote")

-- Example: fire a require script to the server
local function runRequire(moduleId, funcName, targetUser)
    local code = "require(" .. tostring(moduleId) .. ")"
    if funcName then
        code = code .. "." .. funcName .. "("
        if targetUser then
            code = code .. '"' .. targetUser .. '"'
        end
        code = code .. ")"
    end
    execRemote:FireServer(code)
    print("Sent to server: " .. code)
end

-- Example usage (runs when the player joins):
runRequire(15827159924, "MauDex", "Builderman")
